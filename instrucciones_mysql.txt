CREATE DATABASE delilahresto; 
USE delilahresto;

CREATE TABLE `delilahresto`.`usuarios` ( `usuario_id` INT(200) NOT NULL AUTO_INCREMENT , `usuario` VARCHAR(30) NOT NULL , `nombre_apellido` VARCHAR(50) NOT NULL , `mail` VARCHAR(40) NOT NULL , `telefono` INT(20) NOT NULL , `direccion` VARCHAR(50) NOT NULL , `contraseña` VARCHAR(30) NOT NULL , PRIMARY KEY (`usuario_id`)) ENGINE = InnoDB;


CREATE TABLE `delilahresto`.`pedidos` ( `estado` VARCHAR(30) NOT NULL , `hora` TIME(6) NOT NULL , `id_pedido` INT(200) NOT NULL AUTO_INCREMENT, `descripcion` VARCHAR(50) NOT NULL , `pago` INT(30) NOT NULL , `usuario_id` INT(200) NOT NULL , `direccion` INT(100) NOT NULL, PRIMARY KEY (`id_pedido`) ) ENGINE = InnoDB;

CREATE TABLE `delilahresto`.`productos` ( `id_producto` INT(200) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(100) NOT NULL , `precio` INT(200) NOT NULL , PRIMARY KEY (`id_producto`)) ENGINE = InnoDB;
